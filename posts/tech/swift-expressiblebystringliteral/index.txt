<pre><code class="hljs language-swift"><span class="hljs-keyword">let</span> person: <span class="hljs-type">Person</span> <span class="hljs-operator">=</span> <span class="hljs-string">"Thuyen"</span>	<span class="hljs-comment">// Instead of: let person = Person(name: "Thuyen")</span>
